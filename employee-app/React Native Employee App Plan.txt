4. Auth Context

This keeps the employee logged in and loads their profile.

src/context/AuthContext.jsx

import React, { createContext, useEffect, useState } from "react";
import { onAuthStateChanged, signInWithEmailAndPassword, signOut } from "firebase/auth";
import { doc, getDoc } from "firebase/firestore";
import { auth, db } from "../api/firebase";

export const AuthContext = createContext(null);

export function AuthProvider({ children }) {
  const [user, setUser] = useState(null);
  const [employeeProfile, setEmployeeProfile] = useState(null);
  const [loading, setLoading] = useState(true);

  async function loadEmployeeProfile(firebaseUser) {
    const ref = doc(db, "employees", firebaseUser.uid);
    const snap = await getDoc(ref);

    if (!snap.exists()) {
      throw new Error("Employee profile not found.");
    }

    setEmployeeProfile({
      id: snap.id,
      ...snap.data(),
    });
  }

  useEffect(() => {
    const unsubscribe = onAuthStateChanged(auth, async (firebaseUser) => {
      try {
        setUser(firebaseUser);

        if (firebaseUser) {
          await loadEmployeeProfile(firebaseUser);
        } else {
          setEmployeeProfile(null);
        }
      } catch (error) {
        console.error(error);
      } finally {
        setLoading(false);
      }
    });

    return unsubscribe;
  }, []);

  async function login(email, password) {
    await signInWithEmailAndPassword(auth, email, password);
  }

  async function logout() {
    await signOut(auth);
  }

  return (
    <AuthContext.Provider value={{ user, employeeProfile, loading, login, logout }}>
      {children}
    </AuthContext.Provider>
  );
}

src/api/jobs.js

import { collection, query, where, getDocs, doc, updateDoc, serverTimestamp } from "firebase/firestore";
import { db } from "./firebase";

export async function getTodayJobs(companyId, employeeId, todayDateString) {
  const jobsRef = collection(db, "jobs");

  const q = query(
    jobsRef,
    where("companyId", "==", companyId),
    where("assignedEmployeeIds", "array-contains", employeeId),
    where("scheduledDate", "==", todayDateString)
  );

  const snap = await getDocs(q);

  return snap.docs.map((docSnap) => ({
    id: docSnap.id,
    ...docSnap.data(),
  }));
}

export async function updateJobStatus(jobId, status) {
  const ref = doc(db, "jobs", jobId);

  await updateDoc(ref, {
    status,
    updatedAt: serverTimestamp(),
  });
}

8. Job Details Screen
import React, { useEffect, useState } from "react";
import { View, Text, Button } from "react-native";
import { doc, getDoc } from "firebase/firestore";
import { db } from "../api/firebase";
import { updateJobStatus } from "../api/jobs";

export default function JobDetailsScreen({ route, navigation }) {
  const { jobId } = route.params;
  const [job, setJob] = useState(null);

  async function loadJob() {

      <Text style={{ fontSize: 22 }}>{job.serviceType}</Text>
      <Text>{job.address}</Text>
      <Text>Status: {job.status}</Text>
      <Text>Notes: {job.notes || "None"}</Text>

      <Button title="Start Job" onPress={startJob} />
      <Button title="Checklist" onPress={() => navigation.navigate("Checklist", { jobId })} />
      <Button title="Complete Job" onPress={completeJob} />
    </View>
  );
}

import {

  getDocs,
  doc,
  updateDoc,
  serverTimestamp,
} from "firebase/firestore";
import { db } from "./firebase";

export async function getChecklistItems(jobId) {
  const ref = collection(db, "jobs", jobId, "checklistItems");
  const snap = await getDocs(ref);

  return snap.docs.map((docSnap) => ({
    id: docSnap.id,
    ...docSnap.data(),
  }));
}

export async function toggleChecklistItem(jobId, itemId, employeeId, completed) {
  const ref = doc(db, "jobs", jobId, "checklistItems", itemId);

  await updateDoc(ref, {
    completed,
    completedBy: completed ? employeeId : null,
    completedAt: completed ? serverTimestamp() : null,
  });
}

import React, { useContext, useEffect, useState } from "react";

10. Photo Upload
import * as ImagePicker from "expo-image-picker";
import { ref, uploadBytes, getDownloadURL } from "firebase/storage";
import { collection, addDoc, serverTimestamp } from "firebase/firestore";
import { storage, db } from "./firebase";

export async function pickAndUploadJobPhoto(jobId, employeeId, category) {
  const result = await ImagePicker.launchCameraAsync({
    quality: 0.7,
    allowsEditing: false,
  });

  if (result.canceled) return null;

  const asset = result.assets[0];
  const response = await fetch(asset.uri);
  const blob = await response.blob();

  const filePath = `jobPhotos/${jobId}/${Date.now()}.jpg`;
  const fileRef = ref(storage, filePath);

  await uploadBytes(fileRef, blob);

  const downloadUrl = await getDownloadURL(fileRef);

  await addDoc(collection(db, "jobs", jobId, "photos"), {
    employeeId,
    category,
    url: downloadUrl,
    createdAt: serverTimestamp(),
  });

  return downloadUrl;
}

export async function getJobMessages(companyId, jobId) {
  const ref = collection(db, "jobMessages");

  const q = query(
    ref,
    where("companyId", "==", companyId),
    where("jobId", "==", jobId),
    orderBy("createdAt", "asc")

  return snap.docs.map((docSnap) => ({
    id: docSnap.id,
    ...docSnap.data(),
  }));
}
13. Security Rules Concept

Very important.

Employees should only access their company and assigned data.

Example Firestore rule concept:

rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    function isSignedIn() {
      return request.auth != null;
    }

    function employeeDoc() {
      return get(/databases/$(database)/documents/employees/$(request.auth.uid));
    }

    function employeeCompanyId() {
      return employeeDoc().data.companyId;
    }

    match /employees/{employeeId} {
      allow read: if isSignedIn()
        && request.auth.uid == employeeId;

      allow update: if isSignedIn()
        && request.auth.uid == employeeId;
    }

    match /jobs/{jobId} {
      allow read: if isSignedIn()
        && resource.data.companyId == employeeCompanyId()
        && request.auth.uid in resource.data.assignedEmployeeIds;

      allow update: if isSignedIn()
        && resource.data.companyId == employeeCompanyId()
        && request.auth.uid in resource.data.assignedEmployeeIds;
    }
  }
}

Do not launch without tightening security rules.

import React, { createContext, useEffect, useState } from "react";
import { onAuthStateChanged, signInWithEmailAndPassword, signOut } from "firebase/auth";
import { doc, getDoc } from "firebase/firestore";
import { auth, db } from "../api/firebase";

export const AuthContext = createContext(null);

export function AuthProvider({ children }) {
  const [user, setUser] = useState(null);
  const [employee, setEmployee] = useState(null);
  const [loading, setLoading] = useState(true);

  async function loadEmployee(firebaseUser) {
    const employeeRef = doc(db, "employees", firebaseUser.uid);
    const employeeSnap = await getDoc(employeeRef);

    if (!employeeSnap.exists()) {
      throw new Error("Employee profile not found.");
    }

    const data = employeeSnap.data();

    if (data.status !== "active") {
      throw new Error("Employee account is inactive.");
    }

    setEmployee({
      id: employeeSnap.id,
      ...data,
    });
  }

  useEffect(() => {
    return onAuthStateChanged(auth, async (firebaseUser) => {
      try {
        setUser(firebaseUser);

        if (firebaseUser) {
          await loadEmployee(firebaseUser);
        } else {
          setEmployee(null);
        }
      } catch (err) {
        console.error("Auth error:", err);
        setEmployee(null);
      } finally {
        setLoading(false);
      }
    });
  }, []);

  async function login(email, password) {
    await signInWithEmailAndPassword(auth, email.trim(), password);
  }

  async function logout() {
    await signOut(auth);
  }

  return (
    <AuthContext.Provider value={{ user, employee, loading, login, logout }}>
      {children}
    </AuthContext.Provider>
  );
}

import * as ImagePicker from "expo-image-picker";
import { ref, uploadBytes, getDownloadURL } from "firebase/storage";
import { collection, addDoc, serverTimestamp } from "firebase/firestore";
import { storage, db } from "./firebase";

export async function captureAndUploadJobPhoto({
  companyId,
  jobId,
  employeeId,
  category,
}) {
  const permission = await ImagePicker.requestCameraPermissionsAsync();

  if (!permission.granted) {
    throw new Error("Camera permission denied.");
  }

  const result = await ImagePicker.launchCameraAsync({
    quality: 0.7,
    allowsEditing: false,
  });

  if (result.canceled) return null;

  const asset = result.assets[0];
  const response = await fetch(asset.uri);
  const blob = await response.blob();

  const path = `companies/${companyId}/jobs/${jobId}/photos/${Date.now()}.jpg`;
  const fileRef = ref(storage, path);

  await uploadBytes(fileRef, blob);

  const url = await getDownloadURL(fileRef);

  await addDoc(collection(db, "jobs", jobId, "photos"), {
    companyId,
    employeeId,
    category,
    url,
    storagePath: path,
    createdAt: serverTimestamp(),
  });

  return url;
}

13. Tap to Pay With Stripe Terminal

Stripe Terminal Tap to Pay uses a Terminal payment flow: connection token, connect reader, create PaymentIntent, collect payment method, process payment. Stripe documents this as a retryable PaymentIntent-based flow.

Backend Cloud Functions Needed

You need:

createTerminalConnectionToken
createTerminalPaymentIntent
captureTerminalPaymentResult / webhook handling
Cloud Function: Connection Token
const functions = require("firebase-functions");
const admin = require("firebase-admin");
const Stripe = require("stripe");

admin.initializeApp();

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY);

exports.createTerminalConnectionToken = functions.https.onCall(async (data, context) => {
  if (!context.auth) {
    throw new functions.https.HttpsError("unauthenticated", "Login required.");
  }

  const { companyId } = data;

  const companySnap = await admin.firestore()
    .collection("companies")
    .doc(companyId)
    .get();

  if (!companySnap.exists) {
    throw new functions.https.HttpsError("not-found", "Company not found.");
  }

  const company = companySnap.data();

  if (!company.stripeAccountId) {
    throw new functions.https.HttpsError("failed-precondition", "Company has no Stripe account.");
  }

  const token = await stripe.terminal.connectionTokens.create(
    {},
    {
      stripeAccount: company.stripeAccountId,
    }
  );

  return { secret: token.secret };
});
Cloud Function: Create Terminal PaymentIntent

This uses Stripe Connect and your platform fee.

exports.createTerminalPaymentIntent = functions.https.onCall(async (data, context) => {
  if (!context.auth) {
    throw new functions.https.HttpsError("unauthenticated", "Login required.");
  }

  const { companyId, jobId, amount, currency = "usd" } = data;

  if (!amount || amount < 50) {
    throw new functions.https.HttpsError("invalid-argument", "Invalid payment amount.");
  }

  const db = admin.firestore();

  const [companySnap, jobSnap] = await Promise.all([
    db.collection("companies").doc(companyId).get(),
    db.collection("jobs").doc(jobId).get(),
  ]);

  if (!companySnap.exists || !jobSnap.exists) {
    throw new functions.https.HttpsError("not-found", "Company or job not found.");
  }

  const company = companySnap.data();
  const job = jobSnap.data();

  if (job.companyId !== companyId) {
    throw new functions.https.HttpsError("permission-denied", "Job does not belong to company.");
  }

  const platformFee = Math.round(amount * 0.05);

  const paymentIntent = await stripe.paymentIntents.create(
    {
      amount,
      currency,
      payment_method_types: ["card_present"],
      capture_method: "automatic",
      application_fee_amount: platformFee,
      metadata: {
        companyId,
        jobId,
        source: "ServicesOS_employee_app_tap_to_pay",
      },
    },
    {
      stripeAccount: company.stripeAccountId,
    }
  );

  await db.collection("payments").add({
    companyId,
    jobId,
    amount,
    currency,
    platformFee,
    status: "created",
    stripePaymentIntentId: paymentIntent.id,
    source: "tap_to_pay",
    createdBy: context.auth.uid,
    createdAt: admin.firestore.FieldValue.serverTimestamp(),
  });

  return {
    clientSecret: paymentIntent.client_secret,
    paymentIntentId: paymentIntent.id,
  };
});

Payment Button Component

Exact Terminal SDK method names can vary by version, so have your coding assistant verify against installed package docs. The conceptual flow is official: connect Tap to Pay reader, create PaymentIntent, collect, process.

src/components/PaymentButton.jsx

import React, { useState } from "react";
import { Button, Text, View } from "react-native";
import { httpsCallable } from "firebase/functions";
import {
  useStripeTerminal,
} from "@stripe/stripe-terminal-react-native";
import { functions } from "../api/firebase";

export default function PaymentButton({ job, employee, onPaid }) {
  const [status, setStatus] = useState("");

  const {
    initialize,
    discoverReaders,
    connectReader,
    collectPaymentMethod,
    processPayment,
  } = useStripeTerminal();

  async function handleTapToPay() {
    try {
      setStatus("Initializing payment terminal...");
      await initialize();

      setStatus("Finding Tap to Pay reader...");
      const discoverResult = await discoverReaders({
        discoveryMethod: "tapToPay",
        simulated: false,
      });

      if (discoverResult.error) {
        throw new Error(discoverResult.error.message);
      }

      const reader = discoverResult.discoveredReaders?.[0];

      if (!reader) {
        throw new Error("No Tap to Pay reader found.");
      }

      setStatus("Connecting reader...");
      const connectResult = await connectReader({
        reader,
        locationId: job.stripeLocationId,
      });

      if (connectResult.error) {
        throw new Error(connectResult.error.message);
      }

      setStatus("Creating payment...");
      const createIntent = httpsCallable(functions, "createTerminalPaymentIntent");
      const intentResult = await createIntent({
        companyId: employee.companyId,
        jobId: job.id,
        amount: job.balanceDue,
        currency: "usd",
      });

      const { clientSecret } = intentResult.data;

      setStatus("Collecting card...");
      const collectResult = await collectPaymentMethod({
        paymentIntent: clientSecret,
      });

      if (collectResult.error) {
        throw new Error(collectResult.error.message);
      }

      setStatus("Processing payment...");
      const processResult = await processPayment({
        paymentIntent: collectResult.paymentIntent,
      });

      if (processResult.error) {
        throw new Error(processResult.error.message);
      }

      setStatus("Payment complete.");
      await onPaid?.();
    } catch (err) {
      console.error(err);
      setStatus(err.message || "Payment failed.");
    }
  }

  if (!job.balanceDue || job.balanceDue <= 0) {
    return null;
  }

  return (
    <View style={{ marginVertical: 12 }}>
      <Button title={`Collect $${(job.balanceDue / 100).toFixed(2)}`} onPress={handleTapToPay} />
      {status ? <Text>{status}</Text> : null}
    </View>
  );
}

15. Firestore Security Rules

Critical before launch.

Employees should only see:

Their profile
Their assigned jobs
Their assigned training
Messages for their jobs/company

Rule concept:

rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    function signedIn() {
      return request.auth != null;
    }

    function employee() {
      return get(/databases/$(database)/documents/employees/$(request.auth.uid));
    }

    function employeeCompanyId() {
      return employee().data.companyId;
    }

    match /employees/{employeeId} {
      allow read: if signedIn() && request.auth.uid == employeeId;
      allow update: if signedIn() && request.auth.uid == employeeId;
    }

    match /jobs/{jobId} {
      allow read: if signedIn()
        && resource.data.companyId == employeeCompanyId()
        && request.auth.uid in resource.data.assignedEmployeeIds;

      allow update: if signedIn()
        && resource.data.companyId == employeeCompanyId()
        && request.auth.uid in resource.data.assignedEmployeeIds;
    }

    match /jobs/{jobId}/checklistItems/{itemId} {
      allow read, update: if signedIn()
        && get(/databases/$(database)/documents/jobs/$(jobId)).data.companyId == employeeCompanyId()
        && request.auth.uid in get(/databases/$(database)/documents/jobs/$(jobId)).data.assignedEmployeeIds;
    }

    match /jobMessages/{messageId} {
      allow read, create: if signedIn()
        && request.resource.data.companyId == employeeCompanyId();
    }

    match /trainingAssignments/{assignmentId} {
      allow read: if signedIn()
        && resource.data.companyId == employeeCompanyId()
        && resource.data.employeeId == request.auth.uid;
    }

    match /timeEvents/{eventId} {
      allow create: if signedIn()
        && request.resource.data.companyId == employeeCompanyId()
        && request.resource.data.employeeId == request.auth.uid;
    }
  }
}
